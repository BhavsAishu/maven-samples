pipeline {
  agent any 
  label 'maven'

     stages{
  
        stage("Git Checkout"){
         
           steps{
                 
                 script{
                 
                 git branch: 'master', url: 'https://github.com/BhavsAishu/maven-samples.git'
                
                }
           }
        }

        stage("unit test"){
        
           steps{
           container ('maven')
             {
                 
                sh 'mvn clean test'
             }
           
           }
        
        
        }
  
  
     }
}
