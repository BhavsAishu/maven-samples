pipeline {
  agent any 
  tools { 
      maven 'maven' 
      jdk 'java'
      }
     stages{
  
        stage("Git Checkout"){
         
           steps{
                 
                 script{
                 
                 git branch: 'master', url: 'https://github.com/BhavsAishu/maven-samples.git'
                
                }
           }
        }

        stage("unit test"){
        
           steps{
                                    
                sh 'mvn clean test'
             
           
           }
        
        
        }

         stage("Intergration test"){
        
           steps{
                                    
                sh 'mvn verify -DskipUnitTests'
             
           
           }
        
        
        }
  
  
     }
}
